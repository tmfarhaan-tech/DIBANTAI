// --- DATABASE: 100 TOPICS (Tetap sama seperti sebelumnya) --- const topics = [ /* ... isi data topics tetap sama ... */ ]; // --- AUDIO ASSETS --- const alarmSound = new Audio('https://assets.mixkit.co/active_storage/sfx/2869/2869-preview.mp3'); const clickSound = new Audio('https://www.soundjay.com/buttons/sounds/button-16.mp3'); const startSound = new Audio('https://www.soundjay.com/buttons/sounds/button-09.mp3'); // --- APP LOGIC --- let timerSeconds = 120; let timerInterval = null; let history = JSON.parse(localStorage.getItem('dibantai_history')) || []; function init() { renderHistory(); // Memastikan audio ter-load [alarmSound, clickSound, startSound].forEach(s => s.load()); } function generateTopic() { const topicTxt = document.getElementById('topic-text'); const badge = document.getElementById('category-badge'); const container = document.getElementById('display-container'); // 1. Play Sound clickSound.currentTime = 0; clickSound.play(); // 2. Start Animation topicTxt.style.opacity = '0'; container.style.boxShadow = '0 0 30px rgba(99, 102, 241, 0.6)'; container.style.transform = 'scale(0.98)'; setTimeout(() => { const randomTopic = topics[Math.floor(Math.random() * topics.length)]; topicTxt.innerText = randomTopic.text; badge.innerText = randomTopic.cat; // 3. Reset Animation & Fade In topicTxt.style.opacity = '1'; container.style.boxShadow = 'none'; container.style.transform = 'scale(1)'; addToHistory(randomTopic); }, 300); } // --- TIMER LOGIC DENGAN SUARA --- function toggleTimer() { const btn = document.getElementById('timer-toggle'); if (timerInterval) { // PAUSE clearInterval(timerInterval); timerInterval = null; btn.innerText = "Lanjut"; } else { // START/RESUME startSound.currentTime = 0; startSound.play(); // Suara saat timer dimulai btn.innerText = "Pause"; timerInterval = setInterval(() => { if (timerSeconds > 0) { timerSeconds--; updateTimerDisplay(); } else { clearInterval(timerInterval); timerInterval = null; btn.innerText = "Mulai"; alarmSound.play(); // Suara alarm saat habis alert("Waktu Diskusi Habis!"); } }, 1000); } } // --- FUNGSI PENDUKUNG (Tetap sama) --- function addToHistory(topic) { const timestamp = new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' }); history.unshift({ ...topic, time: timestamp }); if (history.length > 5) history.pop(); localStorage.setItem('dibantai_history', JSON.stringify(history)); renderHistory(); } function renderHistory() { const list = document.getElementById('history-list'); list.innerHTML = history.map(item => `
${item.cat} • ${item.time}
${item.text}
`).join(''); } function setTimer(sec) { clearInterval(timerInterval); timerInterval = null; document.getElementById('timer-toggle').innerText = "Mulai"; timerSeconds = sec; updateTimerDisplay(); } function updateTimerDisplay() { const mins = Math.floor(timerSeconds / 60); const secs = timerSeconds % 60; document.getElementById('timer-val').innerText = `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`; } function resetTimer() { clearInterval(timerInterval); timerInterval = null; document.getElementById('timer-toggle').innerText = "Mulai"; timerSeconds = 120; updateTimerDisplay(); } function copyTopic() { const text = document.getElementById('topic-text').innerText; navigator.clipboard.writeText(text); const btn = document.getElementById('copy-text'); btn.innerText = "Copied!"; setTimeout(() => btn.innerText = "Salin Teks", 1500); } window.onload = init;
